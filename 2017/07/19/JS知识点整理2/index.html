      var newObj = {};

          if(typeof obj[key] === "string" || typeof obj[key] === "number" ||
          typeof obj[key] === "boolean" || typeof obj[key] === undefined || 
    typeof obj[key] === null ){
            newObj[key] = obj[key];           //基本类型直接复制
          } else {
            newObj[key] = copy(obj[key]);  //引用类型递归复制

       }
